% needs the trailing slash
\graphicspath{
  {images/}
}

\includegraphics[width=5in]{images/pdf/missingfig.pdf}
\includegraphics[width=5in]{pdf/missingfig.pdf}

% replaces the entire graphics path
\graphicspath{
  {images/pdf/}
}
